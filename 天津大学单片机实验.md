```assembly
  ORG 0000H

  JMP main

  ORG 0003H


  main:

  SETB IT0

  SETB EX0

  SETB EA

  MOV A,#11111110B

  s: RL A

  MOV P2,A

  LCALL delay

  JMP s

delay:mov R7,#0C7H
LOOP1:mov R6,#0FAH
LOOP2:mov R4,#0AH
LOOP3:DJNZ R4,LOOP3
	DJNZ R6,LOOP2
	DJNZ R7, LOOP1
	RET
  END
```

## ⏰实验四 定时实验

```assembly
ORG 	0000H	    ;程序从此地址开始运行
	LJMP 	MAIN	    ;跳转到 MAIN 程序处
    ORG     000BH		;设置中断入口
	LJMP    SUBT
	ORG     0100H
MAIN:
    MOV     sp,#40H
	SETB    EA
	SETB    ET0
	SETB    P1.0
	MOV     R3,#20		  ;设置50ms循环20次即为1s
	MOV     TMOD,#01H	  ;工作方式1
	MOV     DPTR,#15536	  ;设置初值
	MOV     TH0,DPH	  
	MOV     TL0,DPL
	SETB    TR0			   ;启动定时
Here:
    SJMP    $
SUBT:  					  
	DJNZ    R3,NEXT
	CPL     P1.0
	MOV     R3,#20
NEXT:
	MOV     TH0,DPH	  
	MOV     TL0,DPL
	RETI
	END    
```

## 💻实验八 七段译码 

```assembly
ORG 	0000H	    ;程序从此地址开始运行
	LJMP 	MAIN	    ;跳转到 MAIN 程序处

	ORG     003BH   	;延时10ms子程序
DELAY10MS:
    PUSH    psw
	MOV     R6,#20
LOOP1:
    MOV   	R7,#0FAH
LOOP2:
    DJNZ    R7,LOOP2
	DJNZ    R6,LOOP1
	POP     psw
	RET
 
	ORG 	0100H	    ;MAIN 从030H处开始
MAIN:	
  	MOV 	R3,#25
	MOV     R4,#00H
	MOV     R5,#10
    MOV     DPTR,#table
LOOP:
	MOV     A,#08H		  ;位码
	MOV     P2,A
	MOV     A,R4		  ;段码
	MOVC    A,@A+DPTR
	MOV     P0,A
	LCALL   DELAY10MS

	INC     R4
	MOV     A,#04H		  ;位码
	MOV     P2,A
	MOV     A,R4		  ;段码
	MOVC    A,@A+DPTR
	MOV     P0,A
	LCALL   DELAY10MS

	INC     R4
	MOV     A,#02H		  ;位码
	MOV     P2,A
	MOV     A,R4		  ;段码
	MOVC    A,@A+DPTR
	MOV     P0,A
	LCALL   DELAY10MS

	INC     R4
	MOV     A,#01H		  ;位码
	MOV     P2,A
	MOV     A,R4		  ;段码
	MOVC    A,@A+DPTR
	MOV     P0,A
	LCALL   DELAY10MS
	DEC     R4
	DEC     R4
	DEC     R4

	DJNZ    R3,NEXT
    MOV     R3,#25
	INC     R4

	DJNZ    R5,NEXT
    MOV     R5,#10
	MOV     R4,#00H	
NEXT:
   	AJMP    LOOP


table:
    DB     0C0H
	DB     0F9H
	DB     0A4H
	DB     0B0H
	DB     99H
	DB     92H
	DB     82H
	DB     0F8H
	DB     80H
	DB     90H
    DB     0C0H
	DB     0F9H
	DB     0A4H
	END
```
